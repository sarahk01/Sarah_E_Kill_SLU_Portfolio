SELECT
  DATE_TRUNC(run_at_date, Month) Month
  ,'Footprint' as metric
  ,analytics_key
  ,analytics_value
FROM
  `engagement_footprint`
WHERE
  run_at_date IN (
  SELECT
    MAX(run_at_date)
  FROM
    `engagement_footprint`
  GROUP BY
    DATE_TRUNC(run_at_date, Month))

UNION ALL
